Firmware : Pedal-Motor Model

Created by User 9263c, last modified on Nov 25, 2017

Simple model of small motor controlled byÂ NUCLEO-F072RB reading input from potentiometer "pedal".

Purpose: Create simple pedal-motor model to learn about low voltage circuit design and embedded programming.

Project Details

Electrical Components

  • Board: NUCLEO-F072RB
    • MCU: STM32F072RBT6
    • Reads potentiometer position and sends signals to H-Bridge to adjust motor speed/direction.
  • H-bridge: L293NE
    • Switches motor's speed/direction.
    • 9V for motor power
    • 5V for IC logic
    • 3.3V for HI/LO input
  • External Power Supply
    • Powers motors through H-Bridge
    • 9V
  • Motor
  • Rotary Potentiometer
    • Used as input for control of motor's speed/direction.
  • Capacitors
    • Used as

int main(void)
{
  /* USER CODE BEGIN 1 */
  /* USER CODE END 1 */
  /* MCU Configuration----------------------------------------------------------*/
  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();
  /* USER CODE BEGIN Init */
  /* USER CODE END Init */
  /* Configure the system clock */
  SystemClock_Config();
  /* USER CODE BEGIN SysInit */
  /* USER CODE END SysInit */
  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_TIM3_Init();
  MX_ADC_Init();
  MX_USART2_UART_Init();
  /* USER CODE BEGIN 2 */
  //HAL_TIM_Base_Start(&htim3); //starts incorrectly
  HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_2);   
  HAL_ADC_Start(&hadc);
  /* USER CODE END 2 */
  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  HAL_GPIO_WritePin(GPIOC, GPIO_PIN_8, GPIO_PIN_SET);
  HAL_GPIO_WritePin(GPIOC, GPIO_PIN_9, GPIO_PIN_RESET);
  //set_PWM_pulse((uint16_t)(500));
  //print out static text
  // uint8_t* buffer;
  // int len=0; //sprintf will return the length of 'buffer'  
  uint16_t ADC_val = 0; //DO NOT USE B10K
  uint16_t max_ADC_val = 1 << 12; //1 << N NOT 2<<N
  uint16_t variable_pulse_len = 1000; //below ~200us motor will not move with 6V supply
  float pot_pos = 0, prev_pot_pos = 0; //normalized
  while (1)
  {
  /* USER CODE END WHILE */
  /* USER CODE BEGIN 3 */
    //HAL_Delay(10);
    
    if (HAL_ADC_PollForConversion(&hadc, 1000000) == HAL_OK){ //working
      ADC_val = HAL_ADC_GetValue(&hadc); //working in continuous mode
      //sets bottom half of pot rotation as negative, ie remaps (0,1) to (-1,1)
      pot_pos = ( (ADC_val * 1.0f / max_ADC_val) - 0.5) * 2;
      // buffer = (uint8_t*) malloc(16);
      // len=sprintf(buffer,"%i\r\n", (int)(pot_pos*100)); //sprintf will return the length of 'buffer'  
      // HAL_UART_Transmit(&huart2, buffer, len, 1000);
      // free(buffer);
      
      //threshold for direction switch is .1 since the lowest amounts of rotation  will not be enough to run the motor 
      if (pot_pos < 0 && prev_pot_pos >= 0){ 
          HAL_GPIO_WritePin(GPIOC, GPIO_PIN_8, GPIO_PIN_SET);
          HAL_GPIO_WritePin(GPIOC, GPIO_PIN_9, GPIO_PIN_RESET);
      }
      else if (pot_pos > 0 && prev_pot_pos <= 0){ 
          HAL_GPIO_WritePin(GPIOC, GPIO_PIN_8, GPIO_PIN_RESET);
          HAL_GPIO_WritePin(GPIOC, GPIO_PIN_9, GPIO_PIN_SET);
      }
      set_PWM_pulse((uint16_t)( variable_pulse_len * abs(pot_pos) )); //max_pulse_val * ADC_val / max_ADC_val
      prev_pot_pos = pot_pos;
    }
    

  }
  /* USER CODE END 3 */
}

void set_PWM_pulse(uint16_t value)
{
    TIM_OC_InitTypeDef sConfigOC;
  
    sConfigOC.OCMode = TIM_OCMODE_PWM1;
    sConfigOC.Pulse = value;
    sConfigOC.OCPolarity = TIM_OCPOLARITY_HIGH;
    sConfigOC.OCFastMode = TIM_OCFAST_DISABLE;
    HAL_TIM_PWM_ConfigChannel(&htim3, &sConfigOC, TIM_CHANNEL_2);
    HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_2);  
}
